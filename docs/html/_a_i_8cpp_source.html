48int AI::minimax(vector<vector<Piece*>>& board, vector<Piece>& p2Pieces,
49 vector<Piece>& p1Pieces, int gridSize, int depth, bool isMaximizing,
50 bool isPlacementPhase, int alpha, int beta)
51{
52 // Terminal conditions
53 if (depth == 0) {
54 return evaluateBoard(board, isPlacementPhase);
55 }
56

59 return WINNING_SCORE + depth; // Prefer faster wins

62 return LOSING_SCORE - depth; // Prefer slower losses
63 }
64
65 if (isMaximizing) {
66 // AI's turn (Player 2)
67 int maxEval = INT_MIN;

69

75
76 undoMove(board, piece, move.fromRow, move.fromCol, move.toRow, move.toCol);
77
78 maxEval = max(maxEval, eval);
79 alpha = max(alpha, eval);
80

113int AI::evaluateBoard(const vector<vector<Piece*>>& board, bool isPlacementPhase)
114{
115 int score = 0;
116
117 // Score for AI (Player 2)
118 for (int row = 0; row < 5; ++row) {
119 for (int col = 0; col < 5; ++col) {
120 if (board[row][col] && !board[row][col]->isPlayer1()) {
121 int lineCount = 0;
122
123 // Horizontal
124 lineCount = countInLine(board, row, col, 0, 1, false);
125 score += lineCount * lineCount * 10; // Quadratic scoring for longer lines
126
127 // Vertical
128 lineCount = countInLine(board, row, col, 1, 0, false);
129 score += lineCount * lineCount * 10;
130
131 // Diagonal (\‍)
132 lineCount = countInLine(board, row, col, 1, 1, false);
133 score += lineCount * lineCount * 10;
134
135 // Diagonal (/)
136 lineCount = countInLine(board, row, col, 1, -1, false);
137 score += lineCount * lineCount * 10;
138 }
139 }
140 }
141
142 // Subtract score for Player 1
143 for (int row = 0; row < 5; ++row) {
144 for (int col = 0; col < 5; ++col) {
145 if (board[row][col] && board[row][col]->isPlayer1()) {
146 // Check lines of player 1 pieces
147 int lineCount = 0;
148
149 // Horizontal
150 lineCount = countInLine(board, row, col, 0, 1, true);
151 score -= lineCount * lineCount * 10;
152
153 // Vertical
154 lineCount = countInLine(board, row, col, 1, 0, true);
155 score -= lineCount * lineCount * 10;
156
157 // Diagonal (\‍)
158 lineCount = countInLine(board, row, col, 1, 1, true);
159 score -= lineCount * lineCount * 10;
160
161 // Diagonal (/)
162 lineCount = countInLine(board, row, col, 1, -1, true);
163 score -= lineCount * lineCount * 10;
164 }
165 }
166 }

168 return score;

172{

174
175 // Count in positive direction
176 int r = row + dRow;
177 int c = col + dCol;

179 count++;
180 r += dRow;
181 c += dCol;
182 }
183
184 // Count in negative direction
185 r = row - dRow;
