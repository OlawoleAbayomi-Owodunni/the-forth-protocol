95 vector<Piece>& p1Pieces, int gridSize, int depth, bool isMaximizing,
96 bool isPlacementPhase, int alpha, int beta)

109 }

112 // AI's turn (Player 2)

159int AI::evaluateBoard(const vector<vector<Piece*>>& board, bool isPlacementPhase)
160{
161 int score = 0;
162 int gridSize = static_cast<int>(board.size());
163
164 // Strategy-based multipliers
165 int centerMultiplier = 1;
166 int edgeMultiplier = 0;
167 int horizontalMultiplier = 10;
168 int verticalMultiplier = 10;
169 int diagonalMultiplier = 10;
170 int offenseMultiplier = 1;
171 int defenseMultiplier = 1;
172
173 // Adjust multipliers based on strategy
174 switch (m_strategy) {
175 case Strategy::FavorCenter:
176 centerMultiplier = isPlacementPhase ? 8 : 3;
177 break;
178 case Strategy::FavorEdges:
179 centerMultiplier = 0;
180 edgeMultiplier = isPlacementPhase ? 8 : 2;
181 break;
182 case Strategy::FavorDiagonal:
183 diagonalMultiplier = 20;
184 horizontalMultiplier = 8;
185 verticalMultiplier = 8;
186 break;
187 case Strategy::Aggressive:
188 offenseMultiplier = 2;
189 defenseMultiplier = 1;
190 centerMultiplier = isPlacementPhase ? 3 : 1;
191 break;
192 case Strategy::Defensive:
193 offenseMultiplier = 1;
194 defenseMultiplier = 2;
195 centerMultiplier = isPlacementPhase ? 2 : 1;
196 break;
197 case Strategy::Balanced:
198 default:
199 centerMultiplier = isPlacementPhase ? 3 : 1;
200 break;
201 }
202
203 for (int row = 0; row < gridSize; ++row) {
204 for (int col = 0; col < gridSize; ++col) {
205 if (board[row][col]) {
206 int lineCount = 0;
207 bool isEdge = (row == 0 || row == gridSize - 1 || col == 0 || col == gridSize - 1);
208
209 if (!board[row][col]->isPlayer1())
210 {
211 // Position scoring
212 score += scoreCloserToCenter(row, col, gridSize) * centerMultiplier;
213 if (isEdge) score += 5 * edgeMultiplier;
214
215 // Horizontal
216 lineCount = countInLine(board, row, col, 0, 1, false);
217 if (lineCount >= 3) score += lineCount * lineCount * horizontalMultiplier * 15 / 10 * offenseMultiplier;
218 else score += lineCount * lineCount * horizontalMultiplier * offenseMultiplier;
219
220 // Vertical
221 lineCount = countInLine(board, row, col, 1, 0, false);
222 if (lineCount >= 3) score += lineCount * lineCount * verticalMultiplier * 15 / 10 * offenseMultiplier;
223 else score += lineCount * lineCount * verticalMultiplier * offenseMultiplier;
224
225 // Diagonal (\‍)
226 lineCount = countInLine(board, row, col, 1, 1, false);
227 if (lineCount >= 3) score += lineCount * lineCount * diagonalMultiplier * 15 / 10 * offenseMultiplier;
228 else score += lineCount * lineCount * diagonalMultiplier * offenseMultiplier;
229
230 // Diagonal (/)
231 lineCount = countInLine(board, row, col, 1, -1, false);
232 if (lineCount >= 3) score += lineCount * lineCount * diagonalMultiplier * 15 / 10 * offenseMultiplier;
233 else score += lineCount * lineCount * diagonalMultiplier * offenseMultiplier;
234 }
235 else
236 {
237 // Position scoring
238 score -= scoreCloserToCenter(row, col, gridSize) * centerMultiplier;
239 if (isEdge) score -= 5 * edgeMultiplier;
240
241 // Horizontal
242 lineCount = countInLine(board, row, col, 0, 1, true);
243 if (lineCount >= 3) score -= lineCount * lineCount * horizontalMultiplier * 2 * defenseMultiplier;
244 else score -= lineCount * lineCount * horizontalMultiplier * defenseMultiplier;
245
246 // Vertical
247 lineCount = countInLine(board, row, col, 1, 0, true);
248 if (lineCount >= 3) score -= lineCount * lineCount * verticalMultiplier * 2 * defenseMultiplier;
249 else score -= lineCount * lineCount * verticalMultiplier * defenseMultiplier;
250
251 // Diagonal (\‍)
252 lineCount = countInLine(board, row, col, 1, 1, true);
253 if (lineCount >= 3) score -= lineCount * lineCount * diagonalMultiplier * 2 * defenseMultiplier;
254 else score -= lineCount * lineCount * diagonalMultiplier * defenseMultiplier;
255
256 // Diagonal (/)
257 lineCount = countInLine(board, row, col, 1, -1, true);
258 if (lineCount >= 3) score -= lineCount * lineCount * diagonalMultiplier * 2 * defenseMultiplier;

261 }
262 }

268int AI::countInLine(const vector<vector<Piece*>>& board, int row, int col, int dRow, int dCol, bool isPlayer1)
269{
270 int count = 1;
271 int gridSize = static_cast<int>(board.size());
272
273 // Count in positive direction
274 int r = row + dRow;
275 int c = col + dCol;
276 while (r >= 0 && r < gridSize && c >= 0 && c < gridSize && board[r][c] && board[r][c]->isPlayer1() == isPlayer1) {
277 count++;
278 r += dRow;
279 c += dCol;
280 }
281
282 // Count in negative direction
283 r = row - dRow;
284 c = col - dCol;
285 while (r >= 0 && r < gridSize && c >= 0 && c < gridSize && board[r][c] && board[r][c]->isPlayer1() == isPlayer1) {
286 count++;
287 r -= dRow;
288 c -= dCol;
289 }
290
291 return count;

296{

411int AI::scoreCloserToCenter(int row, int col, int gridSize)
412{
413 // Calculate the center position (works for both odd and even grid sizes)
414 float centerRow = (gridSize - 1) / 2.0f;
415 float centerCol = (gridSize - 1) / 2.0f;
416
417 // Calculate Manhattan distance from center
418 float distanceFromCenter = abs(row - centerRow) + abs(col - centerCol);
419
420 // Max possible distance (from corner to center)
421 float maxDistance = centerRow + centerCol;
422
423 // Score inversely proportional to distance (closer = higher score)
424 // Scale to 0-10 points
425 int score = static_cast<int>((1.0f - (distanceFromCenter / maxDistance)) * 10.0f);
426
427 return score;
