208bool Game::checkWinCondition(bool isPlayer1) {
209 // Horizontal
210 for (int row = 0; row < m_gridRows; ++row) {
211 for (int col = 0; col < m_gridCols - 3; ++col) {
212 if (m_board[row][col] && m_board[row][col]->isPlayer1() == isPlayer1 &&
213 m_board[row][col + 1] && m_board[row][col + 1]->isPlayer1() == isPlayer1 &&
214 m_board[row][col + 2] && m_board[row][col + 2]->isPlayer1() == isPlayer1 &&
215 m_board[row][col + 3] && m_board[row][col + 3]->isPlayer1() == isPlayer1) {
216 return true;
217 }
218 }
219 }
220
221 // Vertical
222 for (int col = 0; col < m_gridCols; ++col) {
223 for (int row = 0; row < m_gridRows - 3; ++row) {
224 if (m_board[row][col] && m_board[row][col]->isPlayer1() == isPlayer1 &&
225 m_board[row + 1][col] && m_board[row + 1][col]->isPlayer1() == isPlayer1 &&
226 m_board[row + 2][col] && m_board[row + 2][col]->isPlayer1() == isPlayer1 &&
227 m_board[row + 3][col] && m_board[row + 3][col]->isPlayer1() == isPlayer1) {
228 return true;
229 }
230 }
231 }
232
233 // Diagonal (top-left to bottom-right)
234 for (int row = 0; row < m_gridRows - 3; ++row) {
235 for (int col = 0; col < m_gridCols - 3; ++col) {
236 if (m_board[row][col] && m_board[row][col]->isPlayer1() == isPlayer1 &&
237 m_board[row + 1][col + 1] && m_board[row + 1][col + 1]->isPlayer1() == isPlayer1 &&
238 m_board[row + 2][col + 2] && m_board[row + 2][col + 2]->isPlayer1() == isPlayer1 &&
239 m_board[row + 3][col + 3] && m_board[row + 3][col + 3]->isPlayer1() == isPlayer1) {
240 return true;
241 }
242 }
243 }
244
245 // Diagonal (top-right to bottom-left)
246 for (int row = 0; row < m_gridRows - 3; ++row) {
247 for (int col = 3; col < m_gridCols; ++col) {
248 if (m_board[row][col] && m_board[row][col]->isPlayer1() == isPlayer1 &&
249 m_board[row + 1][col - 1] && m_board[row + 1][col - 1]->isPlayer1() == isPlayer1 &&
250 m_board[row + 2][col - 2] && m_board[row + 2][col - 2]->isPlayer1() == isPlayer1 &&
251 m_board[row + 3][col - 3] && m_board[row + 3][col - 3]->isPlayer1() == isPlayer1) {
252 return true;
253 }
254 }
255 }
256
257 return false;

361void Game::run()
362{
363 sf::Clock clock;
364 sf::Time timeSinceLastUpdate = sf::Time::Zero;
365
366 sf::Time timePerFrame = sf::seconds(1.0f / FPS); // 60 fps
367 while (m_window.isOpen())
368 {
369 processEvents(); // as many as possible
370 timeSinceLastUpdate += clock.restart();
371 while (timeSinceLastUpdate > timePerFrame)
372 {
373 timeSinceLastUpdate -= timePerFrame;
374 processEvents(); // at least 60 fps
375 update(timePerFrame.asMilliseconds()); //60 fps
376#ifdef TEST_FPS
377 x_secondTime += timePerFrame;
378 x_updateFrameCount++;
379 if (x_secondTime.asSeconds() > 1)
380 {
381 std::string updatesPS = "UPS " + std::to_string(x_updateFrameCount - 1);
382 x_updateFPS.setString(updatesPS);
383 std::string drawsPS = "DPS " + std::to_string(x_drawFrameCount);
384 x_drawFPS.setString(drawsPS);
385 x_updateFrameCount = 0;
386 x_drawFrameCount = 0;
387 x_secondTime = sf::Time::Zero;
388 }
389#endif
390 }
391 render(); // as many as possible
392#ifdef TEST_FPS
393 x_drawFrameCount++;
394#endif
395 }
396}

751void Game::executeAIMove()
752{
753 if (m_gamePhase == GamePhase::GameOver) return;
754
755 // Start timing the AI calculation
756 m_aiCalculationClock.restart();
757
758 int searchDepth = 3;
759 switch (m_difficulty) {
760 case Difficulty::Easy:
761 searchDepth = 2;
762 break;
763 case Difficulty::Medium:
764 searchDepth = 3;
765 break;
766 case Difficulty::Hard:
767 searchDepth = 3;
768 break;
769 }
770
771 // In AI vs AI mode, use random placement for variety during placement phase
772 bool useRandomPlacement = m_isAIvsAI && m_gamePhase == GamePhase::Placement;
773
774 // Determine which pieces to use based on current player
775 vector<Piece>& currentPlayerPieces = m_isPlayer1Turn ? m_p1Pieces : m_p2Pieces;
776 vector<Piece>& opponentPieces = m_isPlayer1Turn ? m_p2Pieces : m_p1Pieces;
777
778 // Determine which strategy to use
779 Strategy currentStrategy = Strategy::Balanced;
780 if (m_isAIvsAI) {
781 // AI vs AI: use different strategies for each AI
782 currentStrategy = m_isPlayer1Turn ? m_menu.getAI1Strategy() : m_menu.getAI2Strategy();
783 } else {
784 // PvAI: always use AI strategy for player 2
785 currentStrategy = m_menu.getAIStrategy();
786 }
787
788 // Get the current player's last move (not the opponent's)
789 Move currentPlayerLastMove = m_isPlayer1Turn ? m_lastMoveP1 : m_lastMoveP2;
790
791 Move aiMove = m_ai.findBestMove(m_board, currentPlayerPieces, opponentPieces, m_gridRows,
792 m_gamePhase == GamePhase::Placement, searchDepth, useRandomPlacement, currentPlayerLastMove, currentStrategy);
793
794 // Record calculation time
795 m_lastAICalculationTime = m_aiCalculationClock.getElapsedTime().asMilliseconds();
796
797 if (aiMove.pieceIndex >= 0 && aiMove.pieceIndex < currentPlayerPieces.size()) {
798 m_aiHasMoved = true;
799
800 std::ostringstream oss;
801 oss << "Moves Considered: " << m_ai.getMovesConsidered();
802 m_aiMovesConsideredText.setString(oss.str());
803
804 oss.str("");
805 if (m_gamePhase == GamePhase::Placement) {
806 oss << "Best Move:\nPlace piece at ("
807 << aiMove.toRow << ", " << aiMove.toCol << ")";
808 } else {
809 oss << "Best Move:\nFrom (" << aiMove.fromRow << ", " << aiMove.fromCol
810 << ") to (" << aiMove.toRow << ", " << aiMove.toCol << ")";
811 }
812 m_aiBestMoveText.setString(oss.str());
813
814 oss.str("");
815 oss << "Evaluation Score: " << m_ai.getBestScore();
816 m_aiScoreText.setString(oss.str());
817
818 oss.str("");
819 oss << "Search Depth: " << searchDepth << " levels";
820 m_aiDepthText.setString(oss.str());
821
822 oss.str("");
823 oss << "Calculation Time: " << std::fixed << std::setprecision(1)
824 << m_lastAICalculationTime << "ms";
825 m_aiTimeText.setString(oss.str());
826
827 const float cellSizeXY = 100.0f;
828 const float gridSizeXY = m_gridRows * cellSizeXY;
829 const float x0 = 0.5f * (static_cast<float>(ScreenSize::s_width) - gridSizeXY);
830 const float y0 = 0.5f * (static_cast<float>(ScreenSize::s_height) - gridSizeXY);
831
832 if (m_gamePhase == GamePhase::Movement && aiMove.fromRow >= 0 && aiMove.fromCol >= 0) {
833 Vector2f fromPos = {
834 x0 + static_cast<float>(aiMove.fromCol * cellSizeXY),
835 y0 + static_cast<float>(aiMove.fromRow * cellSizeXY)
836 };
837 m_aiMoveFromIndicator.setPosition(fromPos);
838 } else {
839 m_aiMoveFromIndicator.setPosition({ -1000.0f, -1000.0f });
840 }
841
842 Vector2f toPos = {
843 x0 + static_cast<float>(aiMove.toCol * cellSizeXY),
844 y0 + static_cast<float>(aiMove.toRow * cellSizeXY)
845 };
846 m_aiMoveToIndicator.setPosition(toPos);
847
848 applyAIMove(aiMove);
849 }

853{
854 vector<Piece>& currentPlayerPieces = m_isPlayer1Turn ? m_p1Pieces : m_p2Pieces;
855 Piece* piece = &currentPlayerPieces[move.pieceIndex];
856
857 if (m_gamePhase == GamePhase::Placement) {

859 const float cellSizeXY = 100.0f;

863
864 Vector2f cellPos = {

866 y0 + static_cast<float>(move.toRow * cellSizeXY)
867 };
868 piece->setPosition(cellPos);

870 if (m_isPlayer1Turn) {

872 } else {
873 m_lastMoveP2 = move;

885 x0 + static_cast<float>(move.toCol * cellSizeXY),
886 y0 + static_cast<float>(move.toRow * cellSizeXY)

890 m_lastMove = move;

915void Game::snapToGrid(Vector2f mousePos)
916{
917 if (!m_selectedPiece) return;
918
919 bool foundValidCell = false;
920
921 for (int row = 0; row < m_gridRows; ++row) {
922 for (int col = 0; col < m_gridCols; ++col) {
923 const auto& cell = m_grid[row * m_gridCols + col];
924 if (cell.getGlobalBounds().contains(mousePos))
925 {
926 int prevRow = m_selectedPiece->getGridRow();
927 int prevCol = m_selectedPiece->getGridCol();
928
929 if (m_gamePhase == GamePhase::Placement) {
930 if (placePiece(m_selectedPiece, row, col)) {
931 Vector2f cellPos = cell.getPosition();
932 m_selectedPiece->setPosition(cellPos);
933 foundValidCell = true;
934 endTurn();
935 return;
936 }
937 } else if (m_gamePhase == GamePhase::Movement) {
938 // It's a movement phase
939 if (prevRow >= 0 && prevCol >= 0) {
940 if (movePiece(m_selectedPiece, prevRow, prevCol, row, col)) {
941 Vector2f cellPos = cell.getPosition();
942 m_selectedPiece->setPosition(cellPos);
943 updateBoard();
944 foundValidCell = true;
945 endTurn();
946 return;
947 }
948 }
949 }
950 }
951 }
952 }
953
954 if (!foundValidCell) {
955 m_selectedPiece->restoreOriginalPosition();
956 }
